    Summary#

    actual_hours_costUnitUp#

    Formula

    =IF(
    AND( [Actual (hrs)]@row <> "", [Actual (hrs)]@row <> 0 ),
    [Actual (hrs)]@row,
    SUM( CHILDREN([(@all) ActualHrs]@row) )
)

    budget_hours_all#

    Formula

    =IF(
    AND( [Budget (hrs)]@row <> "", [Budget (hrs)]@row <> 0 ),
    [Budget (hrs)]@row,
    SUM( CHILDREN([(@all) BudgetHrs]@row) )
)

    complete_%_control#

    Formula

    =IF(
    [(@control) EV]@row <> 0,
    [(@control) EV]@row / [(@all) BudgetHrs]@row,
    IF(
        AND( [(@all) BudgetHrs]@row <> 0, [(@all) BudgetHrs]@row <> "" ),
        IFERROR( AVG( CHILDREN([Calculated % Complete]@row) ), "" ),
        ""
    )
)

    cost_code_all#

    Formula

    =IF(
    PARENT([(@all) Cost Code]@row) <> "",
    PARENT([(@all) Cost Code]@row),
    [Cost Code]@row
)

    estimate_at_complete_hours_costUnitUp#

    Formula

    =IF(
    [Actual (hrs)]@row <> "",
    IF(
        [ETC (hrs)]@row <> "",
        [ETC (hrs)]@row,
        [Budget (hrs)]@row
    ) + [(@all) ActualHrs]@row,
    SUM( CHILDREN([(@all) EACHrs]@row) )
)

    planned_value_all#

    Formula

    =IF(
    AND( PV@row <> "", PV@row <> 0 ),
    PV@row,
    SUM( CHILDREN([(@all) PV]@row) )
)

    Calculation#

    baseline_planned_%_complete#

    Formula

    =IF(
    AND([Baseline Start Date]@row <> "", [Baseline Finish Date]@row <> ""),
    ROUND(
        IF(
            TODAY() < [Baseline Start Date]@row,
            0,
            IF(
                TODAY() >= [Baseline Finish Date]@row,
                1,
                NETWORKDAYS(
                    [Baseline Start Date]@row,
                    TODAY()
                ) / NETWORKDAYS( [Baseline Start Date]@row, [Baseline Finish Date]@row )
            )
        ),
    2),
    0
)

    behind_baseline_plan#

    Formula

    =IF( [(@control) % Complete]@row < [Baseline Planned % Complete]@row, 1, 0)

    behind_plan#

    Formula

    =IF( [(@control) % Complete]@row < [Planned % Complete]@row, 1, 0)

    complete_%#

    Formula

    =IF(
    [FUTURE - EVM Method]@row = "Manual % Complete",
    [Manual % Complete]@row,
    IF(
        [FUTURE - EVM Method]@row = "Linearised LoE",
        [FUTURE - Linearised LoE]@row,
        IF(
            [FUTURE - EVM Method]@row = "Gated",
            [FUTURE - Gated Progress Reference]@row,
            IF(
                [FUTURE - EVM Method]@row = "Apportioned Effort",
                "NOT SUPPORTED",
                [Manual % Complete]@row
            )
        )
    )
)

    complete_%#

    Formula

    =IF(
    AND( [Start Date]@row <> "", [Finish Date]@row <> "" ),
    ROUND(
        IF(
            TODAY() < [Start Date]@row,
            0,
            IF(
                TODAY() >= [Finish Date]@row,
                1,
                NETWORKDAYS([Start Date]@row, TODAY()
            ) / NETWORKDAYS( [Start Date]@row, [Finish Date]@row )
        )
    ), 2),
    0
)

    cost_performance_index#

    Formula

    =IFERROR(
    [(@control) EV]@row / [(@all) ActualHrs]@row, 
    ""
)

    cost_variance#

    Formula

    =[(@control) EV]@row - [(@all) ActualHrs]@row

    cost_variance_%#

    Formula

    =IFERROR( 
    CV@row / [(@all) PV]@row, 
    "" 
)

    earned_value_rowCalculation#

    Formula

    =IF(
    SUM(CHILDREN(EV@row)) > 0,
    0,
    [Calculated % Complete]@row * [(@all) BudgetHrs]@row
)

    planned_value#

    Formula

    =IF(
    AND( [Budget (hrs)]@row <> "", [Budget (hrs)]@row <> 0 ),
    [Baseline Planned % Complete]@row * [Budget (hrs)]@row,
    0
)

    resource_loading#

    Formula

    =IFERROR(
    ( [Budget (hrs)]@row / Duration@row)
        /( COUNTM([Resource(s)]@row)
            * IF(
                [Non-Standard Capacity]@row <> "",
                [Non-Standard Capacity]@row,
                8
            )
        ),
    ""
)

    schedule_performance_index#

    Formula

    =IFERROR( 
    [(@control) EV]@row / [(@all) PV]@row, 
    ""
)

    schedule_variance#

    Formula

    =[(@control) EV]@row - [(@all) PV]@row

    schedule_variance_%#

    Formula

    =IFERROR(
    SV@row / [(@all) PV]@row, 
    ""
)

    variance_to_baseline_start_date#

    Formula

    =IF(
    AND( [Baseline Start Date]@row <> "", [Start Date]@row <> "" ),
    NETWORKDAYS( [Start Date]@row, [Baseline Start Date]@row ) - 1,
    ""
)

    variance_to_baseline_finish_date#

    Formula

    =IF(
    AND( [Baseline Finish Date]@row <> "", [Finish Date]@row <> "" ),
    NETWORKDAYS( [Finish Date]@row, [Baseline Finish Date]@row ) - 1,
    ""
)

    wbs#

    Formula

    =IF(
    [WBS Level]@row <= 2,
    Activity@row,
    IF(
        [WBS Level]@row <= 2,
        INDEX(ANCESTORS(Activity@row), 2)
            + IF(
                [WBS Level]@row <= 3,
                "",
                " | " + INDEX(ANCESTORS(Activity@row), 3)
                    + IF(
                        [WBS Level]@row <= 4,
                        "",
                        " | " + INDEX(ANCESTORS(Activity@row), 4)
                    )
            )
    )
)

    wbs_descendants#

    Formula

    =IF( 
    [WBS Level]@row <> 1, 
    COUNT(DESCENDANTS()), 
    SUM(CHILDREN()
)

    wbs_level2_description#

    Formula

    =IF(
    [WBS Level]@row <= 2, Activity@row, 
    INDEX( ANCESTORS(Activity@row), 2 )
)

    Check#

    flag#

    Formula

    =IF(
    COUNT(
        IF( [SETTING - Show Budget Issues]#, [Budget Issues]@row ),
        IF( [SETTING - Show Cost Code Issues]#, [Cost Code Issues]@row ),
        IF( [SETTING - Show Resourcing Issues]#, [Resourcing Issues]@row )
    ) > 0,
    "No",
    IF(
        COUNT(
            IF( [SETTING - Show Budget Issues]#, DESCENDANTS([Budget Issues]@row) ),
            IF( [SETTING - Show Cost Code Issues]#, DESCENDANTS([Cost Code Issues]@row) ),
            IF( [SETTING - Show Resourcing Issues]#, DESCENDANTS([Resourcing Issues]@row) )
        ) > 0,
        "Hold",
        ""
    )
)

    budget_issues#

    Formula

    =IF(
    AND( [Budget (hrs)]@row <> 0, [Budget (hrs)]@row <> "" ),
    IF(
        SUM( CHILDREN([Budget (hrs)]@row) ) > 0,
        "Budget also assigned on children",
        ""
    ),
    ""
)

    cost_code_issues#

    Formula

    =IF(
    [(@all) BudgetHrs]@row <> 0,
    "",
    IF(
        [Cost Code]@row = "",
        IF(
            COUNT(ANCESTORS([Cost Code]@row)) > 1,
            "Multiple ancestor cost codes exist",
            IF(
                AND(
                    COUNT(ANCESTORS([Cost Code]@row)) = 0,
                    [WBS Level]@row <> 1,
                    OR(
                        COUNT(CHILDREN([Cost Code Issues]@row)) > 0,
                        COUNT(CHILDREN(Activity@row)) = 0
                    )
                ),
                "Not cost coded, and no applicable ancestor cost code or successor coding to cover all successors",
                ""
            )
        ),
        IF(
            COUNT(ANCESTORS([Cost Code]@row)) > 0,
            "Cost coded and ancestor cost code(s) also exist",
            ""
        )
    )
)

    resourcing_issues#

    Formula

    ``` =IF( [(@control) % Complete]@row <> 1, IF( AND( COUNTM([Resource(s)]@row) <> 0, [Budget (hrs)]@row <> 0 ), IF( Duration@row <> 0, IF( [Budget (hrs)]@row / Duration@row /( COUNTM([Resource(s)]@row) * IF( [Non-Standard Capacity]@row <> "", [Non-Standard Capacity]@row, 8 ) ) > 1, "Assigned resource(s) overloaded: " + (100 * [Budget (hrs)]@row / Duration@row /( (COUNTM([Resource(s)]@row) * IF( [Non-Standard Capacity]@row <> "", [Non-Standard Capacity]@row, 8 ) ) ) + "%"), "" ), "Duration issue" ), IF( [Budget (hrs)]@row <> 0, "Resource assignment required" ) ) )

    Display/Filter Columns#

    filter_client_view#

    Formula

    =IF(
    OR(
        [Client Hidden?]@row = 1,
        PARENT() = "Hidden"
    ),
    "Hidden",
    "Visible"
)

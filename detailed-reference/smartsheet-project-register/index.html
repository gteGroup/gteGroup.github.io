    Formula

    =IF(
    OR(
        [Value Banding]@row = "High", 
        AND(
            [Value Banding]@row = "Medium", 
            [Risk Banding]@row = "High")
        ), 
        "C1", 
        IF(
            OR(
                AND(
                    [Value Banding]@row = "Medium", 
                    [Risk Banding]@row <> "High"), 
                    AND(
                        [Value Banding]@row = "Low", 
                        [Risk Banding]@row = "High")
            ), 
            "C2", 
            IF(
                AND(
                    [Value Banding]@row = "Low", 
                    [Risk Banding]@row <> "High"
                ), 
                "C3", 
                "Error"
            )
        )
)

    Legacy Potential Status#

    Formula

    =IF(
    OR(
        Status@row = "Active", 
        Status@row = "On-Hold", 
        Status@row = "Complete"), 
        "Won", 
        IF(
            Status@row = "Proposed", 
            IF(
                [Date Issued]@row = "", 
                "Preparing", 
                "Submitted"
            ), 
        IF(
            OR(
                Status@row = "Lost", 
                Status@row = "Cancelled"
            ), 
            Status@row
        )
    )
)

    Legacy Project Status#

    Formula

    =IF(
    OR(
        Status@row = "Active"
        , Status@row = "On-Hold", 
        Status@row = "Complete"
    ), 
    Status@row, 
    ""
)

    Potential Status#

    Formula

    =IF(
    [Legacy Potential Status]@row <> "", 
    [Legacy Potential Status]@row, 
    IF(
        [Proposal Outcome]@row <> "", 
        [Proposal Outcome]@row, 
        IF(
            [Potential Recommendation Approval]@row = "Approved", 
            IF(
                [Proposal Manager Bid/Decline Recommendation]@row = "Decline", 
                "Declined", 
                IF(
                    [Proposal Submitted Date]@row <> "", 
                    "Submitted/Pending", "Preparing"
                )
            ), 
            IF(
                [Potential Recommendation Approval]@row = "Submitted", 
                "Assessing", 
                "New"
            )
        )
    )
)

    Formula

    =IF(
    [Risk Score]@row < [Risk Score Banding - Lower Limit for 'Medium']#, 
    "Low", 
    IF(
        [Risk Score]@row < [Risk Score Banding - Lower Limit for 'High']#,
         "Medium", 
         "High"
    )
)

    Formula

    =IF(
    OR(
        [Project Status]@row = "Complete", 
        [Potential Status]@row = "Declined", 
        [Potential Status]@row = "Cancelled", 
        [Potential Status]@row = "Lost"
    ), 
    "Closed", 
    IF(
        [Project Status]@row <> "", "Project", 
        IF(
            OR(
                [Potential Status]@row = "Submitted/Pending", 
                [Potential Status]@row = "Preparing"
            ), 
            "Proposal", 
            "Potential"
            )
    )
)

    [Budget Hours]@row

    Weighted Score - Capability & Experience#

    Formula

    =VALUE(
    LEFT(
        [Risk - Capability & Experience]@row, 1
    )
) * [Complexity Weighting - Capability & Experience]# / [Complexity Weighting - TOTAL]#

    Weighted Score - Commercial#

    Formula

    =VALUE(
    LEFT(
        [Risk - Commercial]@row, 1
    )
) * [Complexity Weighting - Commercial]# / [Complexity Weighting - TOTAL]#

    Weighted Score - Schedule & Resource Availability#

    Formula

    =VALUE(
    LEFT(
        [Risk - Schedule & Resource Availability]@row, 1
    )
) * [Complexity Weighting - Schedule & Resource Avail.]# / [Complexity Weighting - TOTAL]#

    Weighted Score - Strategic#

    Formula

    =VALUE(
    LEFT(
        [Risk - Strategic]@row, 1
    )
) * [Complexity Weighting - Strategic]# / [Complexity Weighting - TOTAL]#

    Weighted Score - Technical & Reputational#

    Formula

    =VALUE(
    LEFT(
        [Risk - Technical & Reputational]@row, 1
    )
) * [Complexity Weighting - Technical & Reputational]# / [Complexity Weighting - TOTAL]#
